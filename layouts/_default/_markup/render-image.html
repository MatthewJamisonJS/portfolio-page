{{- /* Hugo Responsive Image Render Hook */ -}} {{- /* Converts markdown images to responsive, optimized WebP images with explicit dimensions */ -}} {{- $alt := .Text -}} {{- $caption := "" -}} {{- with .Title -}} {{- $caption = . -}} {{- end -}} {{- /* Try to get image from page resources first, then from static */ -}} {{- $img := .Page.Resources.GetMatch (.Destination | safeURL) -}} {{- $staticPath := "" -}} {{- if not $img -}} {{- /* Image not in page bundle, try static folder */ -}} {{- $staticPath = printf "/static/%s" .Destination -}} {{- $img = resources.Get (.Destination | strings.TrimPrefix "/") -}} {{- end -}} {{- if $img -}} {{- /* Generate responsive image sizes */ -}} {{- $imgSmall := $img.Resize "480x webp q85" -}} {{- $imgMedium := $img.Resize "800x webp q85" -}} {{- $imgLarge := $img.Resize "1200x webp q85" -}} {{- $imgXL := $img.Resize "1600x webp q85" -}} {{- /* WebP sources with srcset for different screen sizes */ -}} {{- /* Fallback img tag with explicit dimensions (prevents CLS) */ -}} {{- if $caption -}} {{- end -}} {{- else -}} {{- /* Fallback: render simple img tag if resource not found */ -}} {{- if $caption -}} {{- end -}} {{- warnf "Image not found as Hugo resource: %s (in %s)" .Destination .Page.File.Path -}} {{- end -}}
